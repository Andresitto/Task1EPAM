using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace Задание_2_EPAM
{
    class Program
    {
        //Функция сортировки
        public static void sortmyArray(int[] myArray)
        {
            int check; //переменная для цикла сортировки
            for (int i = 0; i < myArray.Length - 1; i++)
            {
                for (int j = 0; j < myArray.Length - i - 1; j++)
                {
                    if (myArray[j] > myArray[j + 1])
                    {
                        check = myArray[j];
                        myArray[j] = myArray[j + 1];
                        myArray[j + 1] = check;
                    }
                }
            }
        }
        //Функция проверки наличия заданного числа
        public static void samenumber(int[] myArray,int zchislo)
        {
            for (int i = 0; i < myArray.Length; i++)
            {
                if (zchislo==myArray[i])
                {
                    Console.WriteLine("Заданного числа " + zchislo+" есть в массиве");
                    break;
                }
            }
        }
        static void Main(string[] args)
        {
            {
                //Ввод размера массива 
                Console.WriteLine("Введите размер массива: ");
                int elementscount = int.Parse(Console.ReadLine());
                int[] myArray = new int[elementscount];

                //Ввод массива с клавиатуры
                Console.WriteLine("Введите целые числа в массив");
                for (int i = 0; i < myArray.Length; i++)
                {
                    
                    Console.Write("Введите " + (i + 1) + " число массива: ");
                    int element = int.Parse(Console.ReadLine());
                    myArray[i] = element;
                }
                Console.WriteLine();

                //Сортировананка массива из 1 задания(можно использовать Array.Sort(myArray) вместо написанной функции)
                sortmyArray(myArray);

                //Ввод значения
                Console.WriteLine("Введите целое число, для проверки присутсвия его в массиве");
                int zchislo = int.Parse(Console.ReadLine());

                //Вызов функции
                samenumber(myArray,zchislo);

                Console.WriteLine();
                Console.ReadLine();
            }
        }
    }
}
