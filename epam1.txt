using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace Задание_1_ЕПАМ
//Дан массив целых чисел. Напишите функцию, которая получает данный массив в качестве аргумента и сортирует его по возрастанию, а также программу длядемонстрации работы этой функции.
{
    class Program
    {
        //Функция сортировки(можно использовать Array.Sort(myArray) вместо написанной функции)
        public static void sortmyArray(int[] myArray)
        {
            int check; //переменная для цикла сортировки
            for (int i = 0; i < myArray.Length - 1; i++)
            {
                for (int j = 0; j < myArray.Length - i - 1; j++)
                {
                    if (myArray[j] > myArray[j + 1])
                    {
                        check = myArray[j];
                        myArray[j] = myArray[j + 1];
                        myArray[j + 1] = check;
                    }
                }
            }
        }

        static void Main(string[] args)
        {
            //Ввод размера массива 
            Console.WriteLine("Введите размер массива: ");
            int elementscount = int.Parse(Console.ReadLine());
            int[] myArray = new int[elementscount];

            //Ввод массива с клавиатуры
            for (int i = 0; i < myArray.Length; i++)
            {
                Console.Write("Введите " + (i + 1) + " число массива: ");
                int element = int.Parse(Console.ReadLine());
                myArray[i] = element;
            }
            Console.WriteLine();

            //Вызов функции
            sortmyArray(myArray);
            
            //Вывод всего массива
            Console.WriteLine("Отсортированный массив по возрастанию: ");
            for (int i = 0; i < myArray.Length; i++)
            {
                Console.Write(myArray[i] + " ");
            }
            Console.WriteLine();
        }
    }
}
